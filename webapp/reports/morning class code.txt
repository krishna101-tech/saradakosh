    <script type="module">
        // --- Firebase Imports ---
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, getDoc, setDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        
        // --- App Configuration ---
        const SHEET_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTf9NQNnH7EQodlRZalA0JEaPy91ahzvsAQO1ByUer8m8SbprkopNYucNvJzzLlPjuenXvpexZg7k1B/pub?output=csv';

        // --- App State ---
        let db, auth, userId, settingsDocRef;
        let scheduleConfig = { topics: [], startDate: null, googleMeetLink: '' };
        let clipboardTextWithLinks = '';
        let clipboardTextWithoutLinks = '';

        // --- DOM Elements ---
        const loader = document.getElementById('loader');
        const appContent = document.getElementById('app-content');
        const scheduleContainer = document.getElementById('schedule-container');
        const googleMeetLinkBtn = document.getElementById('google-meet-link-btn');
        const modal = document.getElementById('setup-modal');
        const cancelButton = document.getElementById('cancel-button');
        const saveButton = document.getElementById('save-button');
        const startDateInput = document.getElementById('start-date-input');
        const googleMeetInput = document.getElementById('google-meet-input');
        const alertModal = document.getElementById('alert-modal');
        const alertMessage = document.getElementById('alert-message');
        const alertOkButton = document.getElementById('alert-ok-button');
        
        // --- Firebase Configuration ---
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'srisri-ma-schedule';
        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : { apiKey: "DEMO", authDomain: "DEMO", projectId: "DEMO" };

        const app = initializeApp(firebaseConfig);
        db = getFirestore(app);
        auth = getAuth(app);

        // --- Core Application Logic ---

        /**
         * Fetches and parses the class schedule from the published Google Sheet CSV.
         * @returns {Promise<Array<{name: string, link: string}>>} A promise that resolves to an array of topic objects.
         */
        async function fetchAndParseTopics() {
            try {
                const response = await fetch(SHEET_URL);
                if (!response.ok) {
                    throw new Error(`Network response was not ok: ${response.statusText}`);
                }
                const csvText = await response.text();
                return csvText.split('\n').map(row => row.trim()).filter(row => row).map(row => {
                    const columns = row.split(',').map(col => col.trim());
                    // Remove surrounding quotes if they exist
                    const name = columns[0] ? columns[0].replace(/^"|"$/g, '') : '';
                    const link = columns[1] ? columns[1].replace(/^"|"$/g, '') : '';
                    return { name, link };
                });
            } catch (error) {
                console.error("Failed to fetch or parse Google Sheet:", error);
                showAlert("Could not load the schedule from Google Sheets. Please check the link and your internet connection.");
                return [];
            }
        }

        async function main() {
            onAuthStateChanged(auth, async (user) => {
                if (user) {
                    userId = user.uid;
                    settingsDocRef = doc(db, `artifacts/${appId}/public/data/schedule/settings`);
                    await loadSchedule();
                } else {
                    loader.innerText = "Authentication failed. Please refresh.";
                }
            });

            try {
                const token = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;
                if (token) {
                    await signInWithCustomToken(auth, token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (error) {
                console.error("Error during sign-in process:", error);
                loader.innerText = "Failed to authenticate. Please try again later.";
            }
        }

        async function loadSchedule() {
            const topicsPromise = fetchAndParseTopics();
            const settingsPromise = getDoc(settingsDocRef);

            try {
                const [topics, docSnap] = await Promise.all([topicsPromise, settingsPromise]);
                
                if (topics.length === 0) {
                    loader.innerText = "Failed to load topics from Google Sheet. Cannot display schedule.";
                    return;
                }
                scheduleConfig.topics = topics;

                if (docSnap.exists()) {
                    const data = docSnap.data();
                    scheduleConfig.startDate = data.startDate;
                    scheduleConfig.googleMeetLink = data.googleMeetLink || '#';
                    
                    if (scheduleConfig.startDate) {
                        googleMeetLinkBtn.href = scheduleConfig.googleMeetLink;
                        generateAndDisplaySchedule();
                        appContent.classList.remove('hidden');
                        loader.classList.add('hidden');
                    } else {
                        // Data exists but is incomplete
                        openModal(false);
                    }
                } else {
                    console.log("No settings found. Opening setup modal.");
                    openModal(false);
                }
            } catch (error) {
                 console.error("Error loading schedule settings:", error);
                 loader.innerText = "Error loading schedule settings. Please refresh.";
            }
        }
        
        function generateAndDisplaySchedule() {
            if (!scheduleConfig.topics.length || !scheduleConfig.startDate) return;
            scheduleContainer.innerHTML = '';
            
            const initialStartDate = new Date(scheduleConfig.startDate + 'T00:00:00');
            
            const effectiveDate = new Date();
            if (effectiveDate.getDay() === 6) { // Saturday
                effectiveDate.setDate(effectiveDate.getDate() + 2);
            } else if (effectiveDate.getDay() === 0) { // Sunday
                effectiveDate.setDate(effectiveDate.getDate() + 1);
            }
            effectiveDate.setHours(0, 0, 0, 0);

            const diffTime = effectiveDate.getTime() - initialStartDate.getTime();
            const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));
            const weeksPassed = diffDays >= 0 ? Math.floor(diffDays / 7) : 0;
            const currentTopicIndex = weeksPassed % scheduleConfig.topics.length;
            
            const displayTopics = [...scheduleConfig.topics.slice(currentTopicIndex), ...scheduleConfig.topics.slice(0, currentTopicIndex)];
            
            const currentWeeksMonday = new Date(effectiveDate);
            currentWeeksMonday.setDate(effectiveDate.getDate() - (effectiveDate.getDay() === 0 ? 6 : effectiveDate.getDay() - 1));

            let scheduleHTML = '';
            const baseText = `*Morning Classes*\nMon to Fri @7.45 am\n\n*Google meet*\n${scheduleConfig.googleMeetLink || ''}\n\n`;
            let tempClipboardWithLinks = baseText;
            let tempClipboardWithoutLinks = baseText;

            for (let i = 0; i < displayTopics.length; i++) {
                const topic = displayTopics[i];
                const weekStartDate = new Date(currentWeeksMonday);
                weekStartDate.setDate(weekStartDate.getDate() + i * 7);
                const weekEndDate = new Date(weekStartDate);
                weekEndDate.setDate(weekStartDate.getDate() + 4);
                const dateRange = formatDateRange(weekStartDate, weekEndDate);
                
                const isCurrentWeek = i === 0;
                const cardBg = isCurrentWeek ? 'bg-white' : 'bg-slate-50';
                const shadow = isCurrentWeek ? 'shadow-lg' : 'shadow-sm';
                const border = isCurrentWeek ? 'border-sky-500' : 'border-transparent';
                
                scheduleHTML += `
                    <div class="${cardBg} ${shadow} p-3 rounded-lg border-2 ${border} transition flex flex-col sm:flex-row sm:items-center gap-2 sm:gap-4">
                        <p class="flex-grow min-w-0 text-slate-700" title="${topic.name}">
                            <span class="font-bold text-slate-900">${dateRange}</span> - ${topic.name}
                        </p>
                        ${topic.link 
                            ? `<a href="${topic.link}" target="_blank" rel="noopener noreferrer" class="flex-shrink-0 bg-red-600 hover:bg-red-700 text-white font-bold py-2 px-3 rounded-lg shadow-sm transition-transform transform hover:scale-105 text-sm text-center whitespace-nowrap self-end sm:self-auto">
                                   Playlist
                               </a>` 
                            : '<div class="hidden sm:block sm:w-24 flex-shrink-0"></div>'
                        }
                    </div>
                `;
                
                tempClipboardWithLinks += `*${dateRange}* ${topic.name} ${topic.link ? topic.link : ''}\n\n`;
                tempClipboardWithoutLinks += `*${dateRange}* ${topic.name}\n\n`;
            }

            scheduleContainer.innerHTML = scheduleHTML;
            clipboardTextWithLinks = tempClipboardWithLinks.trim();
            clipboardTextWithoutLinks = tempClipboardWithoutLinks.trim();
        }

        async function handleSave() {
            const startDateValue = startDateInput.value;
            if (!startDateValue) { showAlert("Please select a start date."); return; }
            
            const selectedDate = new Date(startDateValue + 'T00:00:00');
            if (selectedDate.getDay() !== 1) { showAlert("Please select a Monday as the start date."); return; }
            
            const meetLink = googleMeetInput.value.trim();
            if (!meetLink) { showAlert("Please enter a Google Meet link."); return; }

            const newSettings = { startDate: startDateValue, googleMeetLink: meetLink };

            try {
                await setDoc(settingsDocRef, newSettings);
                scheduleConfig.startDate = newSettings.startDate;
                scheduleConfig.googleMeetLink = newSettings.googleMeetLink;
                googleMeetLinkBtn.href = scheduleConfig.googleMeetLink;
                closeModal();
                generateAndDisplaySchedule();
                appContent.classList.remove('hidden');
                loader.classList.add('hidden');
            } catch (error) {
                console.error("Error saving settings:", error);
                showAlert("Could not save the settings. Please check your connection.");
            }
        }

        // --- UI & Event Handlers ---
        
        function openModal(isUpdate = false) {
            if (isUpdate) {
                document.getElementById('modal-title').innerText = "Update Settings";
                document.getElementById('modal-description').innerText = "Edit your Google Meet link or the schedule's initial start date.";
                cancelButton.style.display = 'inline-block';
                startDateInput.value = scheduleConfig.startDate;
                googleMeetInput.value = scheduleConfig.googleMeetLink;
            } else {
                document.getElementById('modal-title').innerText = "Initial Setup";
                document.getElementById('modal-description').innerText = "Please provide the Google Meet link and the start date for the first class in your sheet.";
                cancelButton.style.display = 'none';
                startDateInput.value = '';
                googleMeetInput.value = '';
            }
            modal.style.display = 'flex';
        }

        function closeModal() { modal.style.display = 'none'; }
        
        function showAlert(message) {
            alertMessage.textContent = message;
            alertModal.style.display = 'flex';
        }
        function closeAlert() { alertModal.style.display = 'none'; }

        function copyToClipboard(withLinks = false) {
            const helper = document.getElementById('clipboard-helper');
            helper.value = withLinks ? clipboardTextWithLinks : clipboardTextWithoutLinks;
            helper.select();
            helper.setSelectionRange(0, 99999);
            try {
                document.execCommand('copy');
                showToast(`Schedule ${withLinks ? 'with links' : 'without links'} copied!`);
            } catch (err) {
                console.error('Failed to copy text: ', err);
                showToast("Failed to copy.");
            }
            window.getSelection().removeAllRanges();
        }

        function showToast(message) {
            const toast = document.getElementById('toast');
            toast.textContent = message;
            toast.classList.add('show');
            setTimeout(() => { toast.classList.remove('show'); }, 2500);
        }
        
        document.addEventListener('keydown', (event) => {
            if ((event.ctrlKey || event.metaKey) && (event.key === 'u' || event.key === 'U')) {
                event.preventDefault();
                openModal(true);
            }
        });

        // --- COPY BUTTON CLICK/DBLCLICK LOGIC ---
        const copyButton = document.getElementById('copy-button');
        let clickTimeout = null;
        copyButton.addEventListener('click', () => {
            clearTimeout(clickTimeout);
            clickTimeout = setTimeout(() => {
                copyToClipboard(false);
            }, 200);
        });

        copyButton.addEventListener('dblclick', () => {
            clearTimeout(clickTimeout);
            copyToClipboard(true);
        });
        
        saveButton.addEventListener('click', handleSave);
        cancelButton.addEventListener('click', closeModal);
        alertOkButton.addEventListener('click', closeAlert);

        function formatDateRange(start, end) {
            const monthNames = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];
            const startDay = start.getDate();
            const startMonth = monthNames[start.getMonth()];
            const endDay = end.getDate();
            const endMonth = monthNames[end.getMonth()];

            if (startMonth === endMonth) {
                return `${startDay} - ${endDay} ${startMonth}`;
            } else {
                return `${startDay} ${startMonth} - ${endDay} ${endMonth}`;
            }
        }

        main();
    </script>
